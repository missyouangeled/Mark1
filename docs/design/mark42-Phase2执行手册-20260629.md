```python
"""text_compressor.py 测试群。

覆盖:
  - get_text_compressor() 工厂
  - text_compress(content) 包装函数

设计:
  - 纯函数为主, 不需要 mock
  - 异常输入用 pytest.raises
"""

import pytest

from mark42_modules import text_compressor


class TestTextCompressorFactory:
    """get_text_compressor() 工厂测试群。"""

    def test_factory_returns_instance(self):
        comp = text_compressor.get_text_compressor()
        assert comp is not None
        assert hasattr(comp, "compress")

    def test_factory_singleton(self):
        c1 = text_compressor.get_text_compressor()
        c2 = text_compressor.get_text_compressor()
        assert c1 is c2


class TestTextCompress:
    """text_compress() 包装函数测试群。"""

    def test_compress_normal_text(self, sample_long_text):
        result, meta = text_compressor.text_compress(sample_long_text)
        assert isinstance(result, str)
        assert isinstance(meta, dict)
        assert "ratio" in meta
        assert 0 <= meta["ratio"] <= 1

    def test_compress_empty_text(self):
        result, meta = text_compressor.text_compress("")
        assert result == ""
        assert meta == {} or "skipped" in meta

    def test_compress_repetitive_text(self, sample_repetitive_text):
        result, meta = text_compressor.text_compress(sample_repetitive_text)
        # 重复内容应被压缩
        assert len(result) < len(sample_repetitive_text)
        assert meta.get("ratio", 1.0) < 1.0

    def test_compress_chinese_text(self):
        text = "今天天气很好，适合出去走走。" * 50
        result, meta = text_compressor.text_compress(text)
        assert "今天天气" in result  # 中文不破坏

    def test_metadata_contains_method(self, sample_long_text):
        result, meta = text_compressor.text_compress(sample_long_text)
        assert "method" in meta
        assert meta["method"] in ("rule", "smartcrush", "noop", "skipped")

    def test_metadata_contains_saved_bytes(self, sample_repetitive_text):
        result, meta = text_compressor.text_compress(sample_repetitive_text)
        if meta.get("ratio", 1.0) < 1.0:
            assert meta.get("savedBytes", 0) > 0

    def test_short_text_unchanged(self):
        text = "短"
        result, meta = text_compressor.text_compress(text)
        # 短文本可能跳过压缩
        assert result == text or len(result) <= len(text)

    def test_returns_tuple(self, sample_long_text):
        result = text_compressor.text_compress(sample_long_text)
        assert isinstance(result, tuple)
        assert len(result) == 2
```

```python
import pytest

from mark42_modules import compress_queue


class TestCompressQueue:
    def test_get_singleton_default(self):
        q1 = compress_queue.get_compress_queue()
        q2 = compress_queue.get_compress_queue()
        # 默认 max_workers=2 应该是同对象
        assert q1 is q2

    def test_get_different_workers(self):
        q1 = compress_queue.get_compress_queue(max_workers=1)
        q2 = compress_queue.get_compress_queue(max_workers=4)
        # 不同 workers = 不同对象
        assert q1 is not q2
        # 清理
        compress_queue.shutdown_compress_queue()

    def test_submit_returns_future(self, mocker):
        q = compress_queue.get_compress_queue()
        future = q.submit(lambda x: x * 2, 5)
        assert future is not None
        # 不阻塞: 不调 result.get()
        compress_queue.shutdown_compress_queue()

    def test_shutdown_drains_queue(self):
        q = compress_queue.get_compress_queue()
        # 提交一个任务
        q.submit(lambda: "done")
        # shutdown 应该等任务完成
        compress_queue.shutdown_compress_queue()
        # 不应崩溃

    def test_queue_full_raises(self, mocker):
        q = compress_queue.get_compress_queue(max_workers=1, max_queue_size=1)
        # 提交任务填满队列
        q.submit(lambda: None)
        # 第 2 个应该抛 QueueFull 或类似
        with pytest.raises(Exception):  # QueueFull, Full, 等
            q.submit(lambda: None, block=False)
        compress_queue.shutdown_compress_queue()
```

```python
def pytest_configure(config):
    """注册自定义 marker。"""
    config.addinivalue_line(
        "markers",
        "runslow: 标记需要真生产环境的测试 (默认跳过, 用 --runslow 跑)"
    )


def pytest_collection_modifyitems(config, items):
    """默认跳过 runslow 测试。"""
    if not config.getoption("--runslow", default=False):
        skip_slow = pytest.mark.skip(reason="need --runslow option to run")
        for item in items:
            if "runslow" in item.keywords:
                item.add_marker(skip_slow)


def pytest_addoption(parser):
    """添加 --runslow 命令行选项。"""
    parser.addoption(
        "--runslow",
        action="store_true",
        default=False,
        help="跑 runslow 标记的测试（默认跳过）"
    )
```
